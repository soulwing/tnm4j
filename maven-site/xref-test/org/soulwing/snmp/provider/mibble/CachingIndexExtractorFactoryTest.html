22  import static org.hamcrest.Matchers.is;
23  import static org.hamcrest.Matchers.not;
24  import static org.hamcrest.Matchers.nullValue;
25  import static org.hamcrest.Matchers.sameInstance;

27  import net.percederberg.mibble.Mib;
28  import net.percederberg.mibble.MibLoader;
29  import net.percederberg.mibble.MibValueSymbol;
30  import org.jmock.auto.Mock;
31  import org.jmock.integration.junit4.JUnitRuleMockery;
32  import org.junit.Before;
33  import org.junit.Rule;
34  import org.junit.Test;
35  import org.soulwing.snmp.Formatter;
36  import org.soulwing.snmp.IndexExtractor;
37  
38  /**
39   * Unit tests for {@link CachingIndexExtractorFactory}.
40   *
41   * @author Carl Harris
42   */
43  public class CachingIndexExtractorFactoryTest {
44  
45    @Rule
46    public final JUnitRuleMockery context = new JUnitRuleMockery();
47  
48    private CachingIndexExtractorFactory factory =
49        new CachingIndexExtractorFactory();
50  
51    private Mib mib;
52  
53    @Before
54    public void setUp() throws Exception {
55      MibLoader loader = new MibLoader();
56      mib = loader.load("RFC1213-MIB");
57    }
58  
59    @Test
60    public void testNewIndexExtractor() throws Exception {
61      MibValueSymbol symbol = mib.getSymbolByOid(Constants.IF_DESCR_OID + ".1");
62      IndexExtractor extractor = factory.getIndexExtractor(symbol);
63      assertThat(extractor, is(not(nullValue())));
64      assertThat(factory.getIndexExtractor(symbol), is(sameInstance(extractor)));
65    }
66  
67    @Test
68    public void testNewIndexExtractorForUnknownObject() throws Exception {
69      assertThat(factory.getIndexExtractor(null), is(nullValue()));
70    }
71  
72    @Test(expected = IllegalArgumentException.class)
73    public void testNewIndexExtractorForNonTableColumn() throws Exception {
74      MibValueSymbol symbol = mib.getSymbolByOid(Constants.SYS_DESCR_OID);
75      factory.getIndexExtractor(symbol);
76    }
77  
78  }
