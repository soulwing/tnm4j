19  package org.soulwing.snmp;
20  
21  /**
22   * An object that is registered with an {@link SnmpListener} to receive
23   * SNMP notifications (traps, informs).
24   *
25   * @author Carl Harris
26   */
27  public interface SnmpNotificationHandler {
28  
29    /**
30     * Notifies the recipient that a notification was received by the
31     * associated listener.
32     * @param event an event containing the subject notification
33     * @return {@code true} to indicate that the recipient has handled the
34     *    notification and that other handlers (of lower priority) not yet
35     *    invoked should not be notified
36     */
37    Boolean handleNotification(SnmpNotificationEvent event);
38  
39  }
