18  package org.soulwing.snmp;
19  
20  import java.util.HashMap;
21  import java.util.Map;
22  
23  /**
24   * An abstract base for {@link SnmpTarget} implementations.
25   *
26   * @author Carl Harris
27   */
28  public abstract class SnmpTargetBase implements SnmpTarget {
29  
30    private String address;
31    private int port = SnmpDefaults.AGENT_PORT;
32    private Map<String, Object> properties = new HashMap<String, Object>();
33  
34    /**
35     * {@inheritDoc}
36     */
37    @Override
38    public String getAddress() {
39      return address;
40    }
41  
42    /**
43     * Sets the {@code address} property.
44     * @param address the value to set
45     */
46    public void setAddress(String address) {
47      this.address = address;
48    }
49  
50    /**
51     * {@inheritDoc}
52     */
53    @Override
54    public int getPort() {
55      return port;
56    }
57  
58    /**
59     * Sets the {@code port} property.
60     * @param port the value to set
61     */
62    public void setPort(int port) {
63      this.port = port;
64    }
65  
66    /**
67     * {@inheritDoc}
68     */
69    @Override
70    public Map<String, Object> getProperties() {
71      return properties;
72    }
73  
74    /**
75     * Sets the {@code properties} property.
76     * @param properties the value to set
77     */
78    public void setProperties(Map<String, Object> properties) {
79      this.properties = properties;
80    }
81  
82    @Override
83    public String toString() {
84      return String.format("%s/%d", address, port);
85    }
86  
87  }
