18  package org.soulwing.snmp.provider.mibble;
19  
20  import org.soulwing.snmp.Formatter;
21  
22  
23  class TimeTicksFormatter implements Formatter {
24  
25    private final Formatter delegate;
26    
27    public TimeTicksFormatter(String hint) {
28      this.delegate = hint != null ? 
29          new IntegerFormatter(hint) : new InnerFormatter();
30    }
31  
32    @Override
33    public String format(Object value) {
34      return delegate.format(value);
35    }
36  
37    private static class InnerFormatter implements Formatter {
38      @Override
39      public String format(Object value) {
40        long ticks = ((Number) value).longValue();
41        int hsecs = (int) (ticks % 100);
42        ticks = ticks / 100;
43        int secs = (int) (ticks % 60);
44        ticks = ticks / 60;
45        int mins = (int) (ticks % 60);
46        ticks = ticks / 60;
47        int hours = (int) (ticks % 24);
48        int days = (int) (ticks / 24);
49        return String.format("%dd %02d:%02d:%02d.%02d",
50            days, hours, mins, secs, hsecs);
51      }
52    }
53    
54  }
