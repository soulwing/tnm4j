19  package org.soulwing.snmp.provider.snmp4j;
20  
21  import org.snmp4j.PDU;
22  import org.snmp4j.ScopedPDU;
23  import org.snmp4j.smi.OctetString;
24  import org.soulwing.snmp.SnmpTarget;
25  import org.soulwing.snmp.SnmpV3Target;
26  
27  /**
28   * A {@link PduFactoryStrategy} for an SNMPv2c target.
29   *
30   * @author Carl Harris
31   */
32  class SnmpV3PduFactoryStrategy implements PduFactoryStrategy {
33    
34    /**
35     * {@inheritDoc}
36     */
37    @Override
38    public PduFactory newPduFactory(SnmpTarget target) {
39      if (!(target instanceof SnmpV3Target)) return null;
40      return new SnmpV3PduFactory((SnmpV3Target) target);
41    }
42    
43    static class SnmpV3PduFactory implements PduFactory {
44  
45      private final SnmpV3Target target;
46      
47      /**
48       * Constructs a new instance.
49       * @param target
50       */
51      SnmpV3PduFactory(SnmpV3Target target) {
52        this.target = target;
53      }
54  
55      /**
56       * {@inheritDoc}
57       */
58      @Override
59      public PDU newPDU() {
60        ScopedPDU pdu = new ScopedPDU();
61        if (target.getScope() != null) {
62          pdu.setContextName(new OctetString(target.getScope()));
63        }
64        return pdu;
65      }
66      
67    }
68  
69  }
