19  package org.soulwing.snmp.provider.snmp4j;
20  
21  import org.soulwing.snmp.Formatter;
22  import org.soulwing.snmp.ObjectValue;
23  import org.soulwing.snmp.SnmpTarget;
24  import org.soulwing.snmp.SnmpV1Trap;
25  import org.soulwing.snmp.VarbindCollection;
26  
27  /**
28   * An {@link SnmpV1Trap} implemented on top of SNMP4j.
29   *
30   * @author Carl Harris
31   */
32  class Snmp4jV1Trap extends Snmp4jNotification implements SnmpV1Trap {
33  
34    private String enterprise;
35    private String agentAddress;
36    private ObjectValue genericType;
37    private ObjectValue specificType;
38    private ObjectValue timestamp;
39  
40    public Snmp4jV1Trap(SnmpTarget target, VarbindCollection varbinds) {
41      super(Type.TRAPv1, target, varbinds);
42    }
43  
44    @Override
45    public String getEnterprise() {
46      return enterprise;
47    }
48  
49    public void setEnterprise(String enterprise) {
50      this.enterprise = enterprise;
51    }
52  
53    @Override
54    public String getAgentAddress() {
55      return agentAddress;
56    }
57  
58    public void setAgentAddress(String agentAddress) {
59      this.agentAddress = agentAddress;
60    }
61  
62    @Override
63    public ObjectValue getGenericTrap() {
64      return genericType;
65    }
66  
67    public void setGenericType(ObjectValue genericType) {
68      this.genericType = genericType;
69    }
70  
71    @Override
72    public ObjectValue getSpecificTrap() {
73      return specificType;
74    }
75  
76    public void setSpecificType(ObjectValue specificType) {
77      this.specificType = specificType;
78    }
79  
80    @Override
81    public ObjectValue getTimestamp() {
82      return timestamp;
83    }
84  
85    public void setTimestamp(ObjectValue timestamp) {
86      this.timestamp = timestamp;
87    }
88  
89    @Override
90    public String toString() {
91      return String.format("{ type=%s peer=%s enterprise=%s agent=%s genericType=%s specificType=%s timestamp=%s varbinds=%s }",
92          getType(), getPeer(), enterprise, agentAddress, genericType,
93          specificType, timestamp, getVarbinds());
94    }
95  
96  }
