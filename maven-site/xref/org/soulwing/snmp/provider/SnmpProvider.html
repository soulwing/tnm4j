18  package org.soulwing.snmp.provider;
19  
20  import org.soulwing.snmp.Mib;
21  import org.soulwing.snmp.SnmpContext;
22  import org.soulwing.snmp.SnmpFactoryConfig;
23  import org.soulwing.snmp.SnmpListener;
24  import org.soulwing.snmp.SnmpTarget;
25  import org.soulwing.snmp.SnmpTargetConfig;
26  
27  /**
28   * A provider of {@link SnmpContext} objects.
29   *
30   * @author Carl Harris
31   */
32  public interface SnmpProvider {
33  
34    /**
35     * Gets the provider name.
36     * @return provider name (never {@code null})
37     */
38    String getName();
39  
40    /**
41     * Initializes this provider instance.
42     * <p>
43     * This method is invoked once, after the factory has been instantiated,
44     * before the provider is called upon to create contexts or listeners.
45     * @param config configuration associated with the caling factory
46     */
47    void init(SnmpFactoryConfig config);
48  
49    /**
50     * Creates a new SNMPv2c context.
51     * @param target target agent
52     * @param config context configuration; the provider can safely
53     *    assume that the configuration is immutable for the lifetime of the
54     *    returned context
55     * @param mib MIB provider
56     * @return SNMP context
57     */
58    SnmpContext newContext(SnmpTarget target, SnmpTargetConfig config,
59        Mib mib);
60  
61    /**
62     * Creates a new listener that will receive and distribute SNMP notifications
63     * (traps, informs).
64     * @param address address on which to listen ({@code null} to indicate ANY)
65     * @param port port on which to listen
66     * @param mib MIB provider
67     * @return listener object
68     */
69    SnmpListener newListener(String address, int port, Mib mib);
70  
71    /**
72     * Notifies the recipient that the {@link org.soulwing.snmp.SnmpFactory}
73     * has been closed.
74     * <p>
75     * The provider should release any resources such as threads, locks, open
76     * files/sockets, etc when this method is invoked.
77     */
78    void close();
79  
80  }
