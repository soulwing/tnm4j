19  package org.soulwing.snmp;
20  
21  import java.util.Map;
22  import java.util.concurrent.BlockingQueue;
23  import java.util.concurrent.ConcurrentHashMap;
24  import java.util.concurrent.LinkedBlockingQueue;
25  import java.util.concurrent.TimeUnit;
26  
27  /**
28   * An {@link SnmpCompletionService} that is backed by a {@link BlockingQueue}.
29   *
30   * @author Carl Harris
31   */
32  public class BlockingQueueSnmpCompletionService<V> 
33      implements SnmpCompletionService<V> {
34  
35    private final Map<SnmpCallback<V>, SnmpCallback<V>> pending = 
36        new ConcurrentHashMap<SnmpCallback<V>, SnmpCallback<V>>();
37    
38    private final BlockingQueue<SnmpEvent<V>> queue;
39  
40    /**
41     * Creates a new completion service backed by an unbounded blocking queue.
42     */
43    public BlockingQueueSnmpCompletionService() {
44      this(new LinkedBlockingQueue<SnmpEvent<V>>());
45    }
46  
47    /**
48     * Creates a new completion service backed by the given blocking queue.
49     * @param queue queue that will be used to buffer received response data
50     */
51    public BlockingQueueSnmpCompletionService(BlockingQueue<SnmpEvent<V>> queue) {
52      this.queue = queue;
53    }
54  
55    /**
56     * {@inheritDoc}
57     */
58    @Override
59    public boolean isIdle() {
60      return pending.isEmpty() && queue.isEmpty();
61    }
62  
63    /**
64     * {@inheritDoc}
65     */
66    @Override
67    public SnmpEvent<V> poll() {
68      return queue.poll();
69    }
70  
71    /**
72     * {@inheritDoc}
73     */
74    @Override
75    public SnmpEvent<V> poll(long timeout) throws InterruptedException {
76      return queue.poll(timeout, TimeUnit.MILLISECONDS);
77    }
78  
79    /**
80     * {@inheritDoc}
81     */
82    @Override
83    public SnmpEvent<V> take() throws InterruptedException {
84      return queue.take();
85    }
86  
87    /**
88     * {@inheritDoc}
89     */
90    @Override
91    public void submit(SnmpOperation<V> operation) {
92      SnmpCallback<V> callback = new SnmpCallback<V>() {
93        @Override
94        public void onSnmpResponse(SnmpEvent<V> event) {
95          pending.remove(this);
96          queue.offer(event);
97        } 
98      };
99      
100     pending.put(callback, callback);
101     operation.invoke(callback);
102   }
103 
104 }
