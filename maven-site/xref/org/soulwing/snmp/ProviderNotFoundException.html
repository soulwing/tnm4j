18  package org.soulwing.snmp;
19  
20  /**
21   * An exception thrown when a named provider is not found by the 
22   * {@link ServiceLoader}.
23   *
24   * @author Carl Harris
25   */
26  public class ProviderNotFoundException extends MibException {
27  
28    private static final long serialVersionUID = -6910583914778740034L;
29  
30    private final String providerName;
31    
32    /**
33     * Constructs a new instance.
34     * @param providerName
35     */
36    public ProviderNotFoundException(String providerName) {
37      super("provider not found: " + providerName);
38      this.providerName = providerName;
39    }
40  
41    /**
42     * Gets the name of the provider that was not found.
43     */
44    public String getProviderName() {
45      return providerName;
46    }
47  
48  }
