19  package org.soulwing.snmp;
20  
21  /**
22   * A configuration object for {@link SnmpFactory}.
23   *
24   * @author Carl Harris
25   */
26  public class SnmpFactoryConfig {
27  
28    /**
29     * Default size of the worker pool.
30     */
31    public static final int DEFAULT_WORKER_POOL_SIZE = 4;
32  
33    /**
34     * Default size of the scheduled worker pool.
35     */
36    public static final int DEFAULT_SCHEDULED_WORKER_POOL_SIZE = 1;
37  
38    private int workerPoolSize = DEFAULT_WORKER_POOL_SIZE;
39  
40    private int scheduledWorkerPoolSize = DEFAULT_SCHEDULED_WORKER_POOL_SIZE;
41  
42    /**
43     * Gets the minimum number of threads to keep in the pool for short-lived 
44     * tasks.
45     * <p> 
46     * The specified number of threads will remain available in the pool, even
47     * if idle.  Tasks will queue when all threads are busy. 
48     *
49     * @return number of threads
50     */
51    public int getWorkerPoolSize() {
52      return workerPoolSize;
53    }
54  
55    /**
56     * Sets the minimum number of threads to keep in the pool for short-lived 
57     * tasks.
58     * <p> 
59     * The specified number of threads will remain available in the pool, even
60     * if idle.  Tasks will queue when all threads are busy. 
61     * @param workerPoolSize number of threads
62     */
63    public void setWorkerPoolSize(int workerPoolSize) {
64      this.workerPoolSize = workerPoolSize;
65    }
66  
67    /**
68     * Gets the number of threads to keep in the pool for scheduled tasks.
69     * <p>
70     * The specified number of threads will remain available in the pool, even
71     * if idle.  Tasks that are ready to run (based on configured schedule)
72     * will queue when all threads are busy.
73     * @return number of threads
74     */
75    public int getScheduledWorkerPoolSize() {
76      return scheduledWorkerPoolSize;
77    }
78  
79    /**
80     * Sets the number of threads to keep in the pool for scheduled tasks.
81     * <p>
82     * The specified number of threads will remain available in the pool, even
83     * if idle.  Tasks that are ready to run (based on configured schedule)
84     * will queue when all threads are busy.
85     * 
86     * @param scheduledWorkerPoolSize number of threads
87     */
88    public void setScheduledWorkerPoolSize(int scheduledWorkerPoolSize) {
89      this.scheduledWorkerPoolSize = scheduledWorkerPoolSize;
90    }
91  
92  }
