19  package org.soulwing.snmp;
20  
21  import java.util.HashMap;
22  import java.util.Map;
23  
24  /**
25   * A configuration object for {@link SnmpFactory}.
26   *
27   * @author Carl Harris
28   */
29  public class SnmpFactoryConfig {
30  
31    /**
32     * Default size of the worker pool.
33     */
34    public static final int DEFAULT_WORKER_POOL_SIZE = 4;
35  
36    /**
37     * Default size of the scheduled worker pool.
38     */
39    public static final int DEFAULT_SCHEDULED_WORKER_POOL_SIZE = 1;
40  
41    private final Map<String, Object> properties = new HashMap<String, Object>();
42  
43    private int workerPoolSize = DEFAULT_WORKER_POOL_SIZE;
44  
45    private int scheduledWorkerPoolSize = DEFAULT_SCHEDULED_WORKER_POOL_SIZE;
46  
47    /**
48     * Gets the minimum number of threads to keep in the pool for short-lived 
49     * tasks.
50     * <p> 
51     * The specified number of threads will remain available in the pool, even
52     * if idle.  Tasks will queue when all threads are busy. 
53     *
54     * @return number of threads
55     */
56    public int getWorkerPoolSize() {
57      return workerPoolSize;
58    }
59  
60    /**
61     * Sets the minimum number of threads to keep in the pool for short-lived 
62     * tasks.
63     * <p> 
64     * The specified number of threads will remain available in the pool, even
65     * if idle.  Tasks will queue when all threads are busy. 
66     * @param workerPoolSize number of threads
67     */
68    public void setWorkerPoolSize(int workerPoolSize) {
69      this.workerPoolSize = workerPoolSize;
70    }
71  
72    /**
73     * Gets the number of threads to keep in the pool for scheduled tasks.
74     * <p>
75     * The specified number of threads will remain available in the pool, even
76     * if idle.  Tasks that are ready to run (based on configured schedule)
77     * will queue when all threads are busy.
78     * @return number of threads
79     */
80    public int getScheduledWorkerPoolSize() {
81      return scheduledWorkerPoolSize;
82    }
83  
84    /**
85     * Sets the number of threads to keep in the pool for scheduled tasks.
86     * <p>
87     * The specified number of threads will remain available in the pool, even
88     * if idle.  Tasks that are ready to run (based on configured schedule)
89     * will queue when all threads are busy.
90     * 
91     * @param scheduledWorkerPoolSize number of threads
92     */
93    public void setScheduledWorkerPoolSize(int scheduledWorkerPoolSize) {
94      this.scheduledWorkerPoolSize = scheduledWorkerPoolSize;
95    }
96  
97    /**
98     * Gets the value of a provider-specific property.
99     * @param name name of the value to retrieve
100    * @return property value associated with {@code name} or {@code null} if
101    *    the named property does not exist
102    */
103   public Object getProperty(String name) {
104     return properties.get(name);
105   }
106 
107   /**
108    * Sets the value of a provider-specific property.
109    * @param name name of the property to retrieve
110    * @param value the value to set
111    */
112   public void setProperty(String name, Object value) {
113     properties.put(name, value);
114   }
115 
116 }
