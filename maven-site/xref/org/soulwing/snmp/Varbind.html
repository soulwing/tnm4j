18  package org.soulwing.snmp;
19  
20  public interface Varbind extends ObjectValue {
21  
22    /**
23     * Gets the MIB name of the bound object.
24     * @return name
25     */
26    String getName();
27    
28    /**
29     * Gets the object identifier of the named object as a dotted decimal string.
30     * @return OID string
31     */
32    String getOid();
33    
34    /**
35     * Gets the MIB syntax type identifier of the object.
36     * @return
37     */
38    int getSyntax();
39  
40    /**
41     * Coerces the value of the bound object to a (signed) integer.
42     * @return integer value of the bound object
43     */
44    int asInt();
45    
46    /**
47     * Coerces the value of the bound object to a (signed) long.
48     * @return long value of the bound object
49     */
50    long asLong();
51  
52    /**
53     * Coerces the value of the bound object to a display string.
54     * <p>
55     * The implementation will incorporate the application textual convention
56     * and display hint (if available) of the corresponding MIB object
57     * definition.
58     * @return string representation of the bound object
59     */
60    String asString();
61    
62    /**
63     * Gets the value of the object as one of the fundamental types of SMI.
64     * <p>
65     * This method returns a value for each SMI type as follows:
66     * <ul>
67     * <li>OBJECT IDENTIFIER &mdash; {@code int} array of the OID components</li>
68     * <li>OCTET STRING &mdash; {@code byte} array containing the object's octets</li>
69     * <li>INTEGER &mdash; {@code Long} value of the object</li>
70     * <li>NULL &mdash; {@code null}</li>
71     * <li>IpAddress &mdash; {@code byte} array containing the address's octets</li>
72     * <li>Counter &mdash; {@code Long} value of the object</li>
73     * <li>Gauge &mdash; {@code Long} value of the object</li>
74     * <li>TimeTicks &mdash; {@code Long} value of the object</li>
75     * <li>Opaque &mdash; {@code byte} array containing the object's octets</li>
76     * </ul>
77     * @return
78     */
79    Object toObject();
80    
81    /**
82     * Gets the bindings that represent the index values the bound value.
83     * @return if the bound value is a member of a conceptual table, the
84     *    variable bindings representing the object's table index are returned;
85     *    an empty array is returned for an object that is not a table member
86     */
87    Varbind[] getIndexes();
88    
89  }
