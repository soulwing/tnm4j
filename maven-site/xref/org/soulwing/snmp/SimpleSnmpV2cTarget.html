18  package org.soulwing.snmp;
19  
20  
21  /**
22   * A {@link SnmpV2cTarget} implemented as a simple configurable bean.
23   *
24   * @author Carl Harris
25   */
26  public class SimpleSnmpV2cTarget extends SnmpTargetBase 
27      implements SnmpV2cTarget {
28  
29    private String community;
30    
31    /**
32     * {@inheritDoc}
33     */
34    @Override
35    public String getCommunity() {
36      return community;
37    }
38  
39    /**
40     * Sets the {@code community} property.
41     * @param community the value to set
42     */
43    public void setCommunity(String community) {
44      this.community = community;
45    }
46  
47  }
