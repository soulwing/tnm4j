19  package org.soulwing.snmp;
20  
21  /**
22   * An SNMP operation.
23   *
24   * @author Carl Harris
25   */
26  public interface SnmpOperation<V> {
27  
28    /**
29     * Invoke the operation on the remote agent and wait for the response.
30     * <p>
31     * This method will block until a response becomes available.
32     * @return response object
33     */
34    SnmpResponse<V> invoke();
35    
36    /**
37     * Invoke this operation, notifying the specified callback when a 
38     * response is available.
39     * @param callback callback that will be notified when a response is
40     *    available
41     */
42    void invoke(SnmpCallback<V> callback);
43    
44  }
